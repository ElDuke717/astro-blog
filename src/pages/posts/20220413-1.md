```javascript
// Here's a fun Regex example that finds smileys in an array.

const faceArray = [";D", ":-(", ":-)", ";~)"]

const countSmileys = (arr) => 
	    arr
      .map(x=> /[;|:][-|~]?[\)|D]/gm.test(x))
      .filter(x=> x === true).length


countSmileys(faceArray) //3

```
